\section{Algorithmic Structure}

Let the primary velocity-pressure state be $U_h = (\boldsymbol{u}_h, p_h) \in \mathcal{X}_h$. The continuous sub-grid projection variables are $\boldsymbol{\pi}_h = (\boldsymbol{\pi}_{h,u}, \pi_{h,p})^\top \in \mathcal{X}_h$. 

To natively balance the numerical disparity between the momentum and mass conservation equations without \textit{a priori} macroscopic scaling, we extract an element-wise scaling vector from the exact Jacobian at each iteration: $w_k = \max( |[\mathcal{J}]_{kk}|, \epsilon_{\text{mach}} \|\text{diag}(\mathcal{J})\|_\infty, \epsilon_{\text{mach}} )$. The diagonally equilibrated dimensionless merit function is defined as:
\begin{equation}
    \Phi_{\boldsymbol{w}}(\boldsymbol{x}) = \frac{1}{2} \sum_{k=1}^N \left( \frac{b_k(\boldsymbol{x})}{w_k} \right)^2
\end{equation}

\begin{algorithm}[H]
\caption{\textbf{Algorithm A:} $\operatorname{ExactNewtonPipeline}$}
\label{alg:A}
\SetAlgoLined
\textbf{Numerical Parameters:} $\tau_{\text{ftol}}, N_{\text{inner}}, N_{\text{inc}}, \tau_{\text{stagnation}}, \tau_{\text{xtol}}$ \\
\KwIn{Initial geometric fields $\boldsymbol{x}^0 \in \mathcal{X}_h$}
\KwOut{$(S_{\text{success}} \in \{\text{True}, \text{False}\}, \; I_{\text{iters}} \in \mathbb{N}, \; \|\boldsymbol{b}\|_\infty \in \mathbb{R})$}
\vspace{0.2cm}
\hrule
\vspace{0.2cm}
Initialize $\boldsymbol{b}_{\text{best}} = \|\boldsymbol{b}(\boldsymbol{x}^0)\|_\infty, I_{\text{up}} = 0$\;
\If{$\|\boldsymbol{b}(\boldsymbol{x}^0)\|_\infty \leq \tau_{\text{ftol}}$}{ \Return \textsf{Success} (Exit: \textsf{Initial F-Tol}) }

\For{$i = 1, \dots, N_{\text{inner}}$}{
    Compute continuous variables: $\mathcal{J}_i, \boldsymbol{b}_i, \boldsymbol{w}, \Phi_{\text{curr}}$\;
    $S_{\text{fail}}, \delta\boldsymbol{x} \gets \operatorname{SolveLinearSystem}(\mathcal{J}_i, \boldsymbol{b}_i)$\;
    \If{$S_{\text{fail}}$}{
        Restore $\boldsymbol{x} \gets \boldsymbol{x}_{\text{best}}$. \Return $(S_{\text{conv}}=\text{False}, I_{\text{iters}}=i, \|\boldsymbol{b}_{\text{best}}\|_\infty)$\;
    }
    
    $S_{\text{step}}, \alpha, L_{\text{step}}, \hat{\boldsymbol{b}}, \hat{\Phi} \gets \operatorname{ArmijoLinesearch}(\boldsymbol{x}, \boldsymbol{x}_{\text{old}}, \boldsymbol{b}, \delta\boldsymbol{x}, \Phi_{\text{curr}})$\;
    
    \If{\textnormal{\textbf{not}} $S_{\text{step}}$ \textnormal{ \textbf{or} } $\|\hat{\boldsymbol{b}}\|_\infty < \boldsymbol{b}_{\text{best}}$}{
        $\boldsymbol{x}_{\text{best}} = \boldsymbol{x}, \boldsymbol{b}_{\text{best}} = \|\hat{\boldsymbol{b}}\|_\infty$\;
    }
    
    $R_{\text{stop}}, S_{\text{conv}}, S_{\text{break}}, I_{\text{up}} \gets \operatorname{EvaluateSafeguards}(i, S_{\text{step}}, L_{\text{step}}, \hat{\boldsymbol{b}}, \hat{\Phi}, \Phi_{\text{curr}})$\;
    \If{$S_{\text{break}}$}{
        \If{\textnormal{\textbf{not}} $S_{\text{conv}}$}{ Restore $\boldsymbol{x} \gets \boldsymbol{x}_{\text{best}}$ }
        \Return $(S_{\text{conv}}, i, \|\boldsymbol{b}_{\text{best}}\|_\infty)$\;
    }
}
\end{algorithm}

\begin{algorithm}[H]
\caption{\textbf{Algorithm A.1:} $\operatorname{SolveLinearSystem}$}
\label{alg:A1}
\SetAlgoLined
\textbf{Numerical Parameters:} (None) \\
\KwIn{Sparse continuous Jacobian $\mathcal{J}_i$, Residual array $\boldsymbol{b}_i$}
\KwOut{$(S_{\text{fail}} \in \{\text{True}, \text{False}\}, \; \delta\boldsymbol{x} \in \mathbb{R}^N)$}
\vspace{0.2cm}
\hrule
\vspace{0.2cm}
\textbf{try} \{ \\
\quad Solve linear system functionally: $\mathcal{J}_i \delta\boldsymbol{x} = \boldsymbol{b}_i$\;
\} \textbf{catch} \{ \\
\quad \Return $(\text{True}, \delta\boldsymbol{x}_{\text{undefined}})$ \tcp*{\textsf{Linear Solve Exception}}
\} \\
\If{$\delta\boldsymbol{x}$ contains NaNs}{
    \Return $(\text{True}, \delta\boldsymbol{x}_{\text{undefined}})$ \tcp*{\textsf{Catastrophic Math Failure}}
}
\Return $(\text{False}, \delta\boldsymbol{x})$ \tcp*{\textsf{Linear System Bounded}}
\end{algorithm}

\begin{algorithm}[H]
\caption{\textbf{Algorithm A.2:} $\operatorname{ArmijoLinesearch}$}
\label{alg:A2}
\SetAlgoLined
\textbf{Numerical Parameters:} $\alpha_{\text{min}}, c_1, \gamma, N_{\text{ls\_max}}$ (extracted from $\Theta_{\text{params}}$) \\
\KwIn{State fields $\boldsymbol{x}, \boldsymbol{x}_{\text{old}}, \boldsymbol{b}$, step mapping $\delta\boldsymbol{x}$, exact directional derivative $D$, current merit $\Phi_{\text{curr}}$}
\KwOut{$(S_{\text{accepted}} \in \{\text{True}, \text{False}\}, \; \alpha \in \mathbb{R}^+, \; L_{\text{step}} = \|\alpha \delta\boldsymbol{x}\| \in \mathbb{R}^+, \; \|\hat{\boldsymbol{b}}\|_\infty \in \mathbb{R}, \; \hat{\Phi}_{\boldsymbol{w}} \in \mathbb{R})$}
\vspace{0.2cm}
\hrule
\vspace{0.2cm}
Initialize $\alpha = 1.0, S_{\text{step\_accepted}} = \text{false}$\;
\For{$j = 1, \dots, N_{\text{ls\_max}}$}{
    $\boldsymbol{x}_{\text{trial}} = \boldsymbol{x}_{\text{old}} - \alpha \delta\boldsymbol{x}$\;
    Compute $\hat{\boldsymbol{b}} = \boldsymbol{b}(\boldsymbol{x}_{\text{trial}})$ and $\hat{\Phi}_{\boldsymbol{w}} = \Phi_{\boldsymbol{w}}(\boldsymbol{x}_{\text{trial}})$\;
    \If{\textnormal{\textbf{not}} $\text{isnan}(\hat{\Phi}_{\boldsymbol{w}})$ \textnormal{\textbf{and}} $\hat{\Phi}_{\boldsymbol{w}} \leq \Phi_{\text{curr}} + c_1 \alpha (-2\Phi_{\text{curr}})$}{
        $S_{\text{step\_accepted}} = \text{true}$. \textbf{break}\;
    }
    $\alpha = \gamma \alpha$\;
    \If{$\alpha \leq \alpha_{\text{min}}$}{\textbf{break}\;}
}
\Return $(S_{\text{step\_accepted}}, \; \alpha, \; \|\alpha \delta\boldsymbol{x}\|, \; \|\hat{\boldsymbol{b}}\|_\infty, \; \hat{\Phi}_{\boldsymbol{w}})$\;
\end{algorithm}

\begin{algorithm}[H]
\caption{\textbf{Algorithm A.3:} $\operatorname{EvaluateSafeguards}$}
\label{alg:A3}
\SetAlgoLined
\textbf{Numerical Parameters:} $\tau_{\text{ftol}}, \tau_{\text{stagnation}}, \tau_{\text{xtol}}, N_{\text{inner}}, N_{\text{inc}}, F_{\text{div}}$ (extracted from $\Theta_{\text{params}}$) \\
\KwIn{Iteration $i$, Validation $S_{\text{step\_accepted}}$, step norm $L_{\text{step}}$, current residual $\|\hat{\boldsymbol{b}}\|_\infty$, scalar merit $\hat{\Phi}_{\boldsymbol{w}}$, base merit $\Phi_{\text{curr}}$}
\KwOut{$(R_{\text{stop}}, \; S_{\text{conv}} \in \{\text{True}, \text{False}\}, \; S_{\text{break}} \in \{\text{True}, \text{False}\}, \; I_{\text{up}} \in \mathbb{N})$}
\vspace{0.2cm}
\hrule
\vspace{0.2cm}
\If{$\|\hat{\boldsymbol{b}}\|_\infty \leq \tau_{\text{ftol}}$}{ \Return \textsf{F-Tol Reached}, \texttt{true}, \texttt{true}, $I_{\text{up}}$\; }
\If{\textnormal{\textbf{not}} $S_{\text{step\_accepted}}$}{ 
    \If{$\|\hat{\boldsymbol{b}}\|_\infty \leq \tau_{\text{stagnation}}$}{ \Return \textsf{Noise Floor Stagnation}, \texttt{true}, \texttt{true}, $I_{\text{up}}$\; }
    \Return \textsf{Linesearch Failed}, \texttt{false}, \texttt{true}, $I_{\text{up}}$\; 
}
\If{$\hat{\Phi}_{\boldsymbol{w}} > \Phi_{\text{curr}} \cdot F_{\text{div}}$}{
    $I_{\text{up}} = I_{\text{up}} + 1$\;
    \If{$I_{\text{up}} \geq N_{\text{inc}}$}{ \Return \textsf{Merit Divergence Escaped}, \texttt{false}, \texttt{true}, $I_{\text{up}}$\; }
} \Else { $I_{\text{up}} = 0$\; }
\If{$L_{\text{step}} \leq \tau_{\text{xtol}}$}{
    \If{$\|\hat{\boldsymbol{b}}\|_\infty \leq \tau_{\text{stagnation}}$}{ \Return \textsf{Noise Floor Stagnation}, \texttt{true}, \texttt{true}, $I_{\text{up}}$\; }
    \Return \textsf{X-Tol Stagnation}, \texttt{false}, \texttt{true}, $I_{\text{up}}$\; 
}
\If{$i == N_{\text{inner}}$}{
    \If{$\|\hat{\boldsymbol{b}}\|_\infty \leq \tau_{\text{stagnation}}$}{ \Return \textsf{Noise Floor Stagnation}, \texttt{true}, \texttt{true}, $I_{\text{up}}$\; }
    \Return \textsf{Iteration Cap Stagnation}, \texttt{false}, \texttt{true}, $I_{\text{up}}$\; 
}
\Return \textsf{In-Progress}, \texttt{false}, \texttt{false}, $I_{\text{up}}$\;
\end{algorithm}

\begin{algorithm}[H]
\caption{\textbf{Algorithm B:} $\operatorname{PicardFallback}$}
\label{alg:B}
\SetAlgoLined
\textbf{Numerical Parameters:} $\tau_{\text{ftol}}, \tau_{\text{stagnation}}$ \\
\KwIn{Initial field guess $\boldsymbol{x}^0 \in \mathcal{X}_h$, Preconditioner backup $\boldsymbol{x}^0_{\text{backup}} \in \mathcal{X}_h$}
\KwOut{$(S_{\text{success}} \in \{\text{True}, \text{False}\}, \; I_{\text{iters}} \in \mathbb{N}, \; \|\boldsymbol{b}\|_\infty \in \mathbb{R})$}
\vspace{0.2cm}
\hrule
\vspace{0.2cm}
Configure parameters zeroing sub-grid scales: $\boldsymbol{\pi}_h = \mathbf{0}$\;
Attempt Newton resolution: $S_{\text{N}}, \dots \gets \operatorname{ExactNewtonPipeline}(\boldsymbol{x}^0)$ targeting $\tau_{\text{ftol}}$\;
\If{$S_{\text{N}}$ formally aborts (Exit via \textsf{Linesearch Failed} or \textsf{Merit Divergence Escaped})}{
    Evaluate strictly guarded Picard Homotopy pass smoothing fields towards stable continuous initial geometry quadratic basins\;
    Re-evaluate Newton sequence definitively mapping cleanly: $S_{\text{success}}, I_{\text{iters}}, \|\boldsymbol{b}\|_\infty \gets \operatorname{ExactNewtonPipeline}(\boldsymbol{x}^0_{\text{smoothed}})$\;
    \If{Terminal evaluation fundamentally bypasses both \textsf{F-Tol Reached} and \textsf{Noise Floor Stagnation} structures}{
        \Return $(\text{False}, I_{\text{iters}}, \|\boldsymbol{b}\|_\infty)$ \tcp*{\textsf{Catastrophic Algebraic Sequence Stagnation}}
    }

\Return $(\text{True}, I_{\text{iters}}, \|\boldsymbol{b}\|_\infty)$ \tcp*{\textsf{Mathematically Locked Globally Bounded State}}
\end{algorithm}

\begin{algorithm}[H]
\caption{\textbf{Algorithm C:} $\operatorname{OSGSFractionalRelaxation}$}
\label{alg:C}
\SetAlgoLined
\textbf{Numerical Parameters:} $N_{\text{OSGS}}, m_{\text{warmup}}, \tau_{\text{warmup}}, \tau_{\text{ftol}}, \tau_{\text{OSGS}}, \tau_{\text{proj}}, \tau_{\text{stagnation}}$ \\
\KwIn{Converged steady state $U_h^{\text{ASGS}} \in \mathcal{X}_h$, Algebraic forms $(\mathcal{R}_u, \mathcal{R}_p)$}
\KwOut{$(S_{\text{success}} \in \{\text{True}, \text{False}\}, \; \boldsymbol{\pi}_{h,u} \in \boldsymbol{V}_h, \; \pi_{h,p} \in Q_h, \; I_{\text{iters}} \in \mathbb{N}, \; \|\boldsymbol{b}\|_\infty \in \mathbb{R})$}
\vspace{0.2cm}
\hrule
\vspace{0.2cm}

Bootstrap initial sub-grid projections:
$\boldsymbol{\pi}_{h,u}^0 = \Pi_{\boldsymbol{V}}(\mathcal{R}_{u}(U_h^0))$ and $\pi_{h,p}^0 = \Pi_{Q}(\mathcal{R}_p(U_h^0))$\;
\For{$m = 1, \dots, N_{\text{OSGS}}$}{
    Set inner tolerance limit: $\tau_{\text{inner}}^m = (m \leq m_{\text{warmup}}) \; ? \; \max(\tau_{\text{ftol}}, \tau_{\text{warmup}}) : \tau_{\text{ftol}}$\;
    
    Set $\boldsymbol{\pi}_h = \boldsymbol{\pi}_h^{m-1}$ fixed, resolve continuous algebraic system executing strict fraction mapping via $\operatorname{ExactNewtonPipeline}$ targeting limited iterations $N_{\text{inner}}$\;
    \If{Monolithic sweep returns algebraic collapse natively \textnormal{(}\textsf{Linesearch Failed} \textnormal{\textbf{or}} \textsf{Merit Divergence Escaped}\textnormal{)}}{
        Abort structural iteration immediately (mathematically divergent algebraic boundaries violated)\;
    }
    
    \tcc{Evaluate geometric state fraction shift explicitly evaluating explicit mass vectors linearly:}
    $d_{\text{state}}^m = \max \left( \| \boldsymbol{u}_h^m - \boldsymbol{u}_h^{m-1} \|_{L^\infty(\Omega)}, \; \| p_h^m - p_h^{m-1} \|_{L^\infty(\Omega)} \right)$\;
    
    Update continuous evaluation geometries arrays natively: $\boldsymbol{\mathcal{R}}_u^m = \mathcal{R}_u(U_h^m)$ and $\mathcal{R}_p^m = \mathcal{R}_p(U_h^m)$\;
    Update continuous orthogonal projections:
    $(\boldsymbol{\pi}_{h,u}^m, \pi_{h,p}^m) = \left( \Pi_{\boldsymbol{V}}(\boldsymbol{\mathcal{R}}_u^m), \Pi_Q(\mathcal{R}_p^m) \right)$\;
    
    \tcc{Independent Block-Weighted Anderson Acceleration}
    \textit{(Optional)} Filter limits through separated bounds using $L^2$ mass matrices:
    \begin{align*}
        \boldsymbol{\pi}_{h,u}^m &\gets \text{Anderson}( \text{History}_u, \boldsymbol{\pi}_{h,u}^{m-1}, \boldsymbol{\pi}_{h,u}^m, M_u ) \\
        \pi_{h,p}^m &\gets \text{Anderson}( \text{History}_p, \pi_{h,p}^{m-1}, \pi_{h,p}^m, M_p )
    \end{align*}
    
    \tcc{Define orthogonal projection drift explicitly evaluating algebraic block arrays:}
    $d_\pi^m = \max \left( \sqrt{(\delta\boldsymbol{\pi}_{h,u}^m)^\top M_u \delta\boldsymbol{\pi}_{h,u}^m}, \; \sqrt{(\delta\pi_{h,p}^m)^\top M_p \delta\pi_{h,p}^m} \right)$\;
    
    \If{$\left( d_{\text{state}}^m \leq \max(\tau_{\text{OSGS}}, \tau_{\text{stagnation}}) \right)$ \textnormal{\textbf{AND/OR}} $\left( d_\pi^m \leq \max(\tau_{\text{proj}}, \tau_{\text{stagnation}}) \right)$}{
        \Return \texttt{true}, $\boldsymbol{\pi}_{h,u}^m, \pi_{h,p}^m, I_{\text{iters}}, \|\boldsymbol{b}\|_\infty$ \tcp*{\textsf{Validated Staggered Boundary}}
    }
}
\Return \texttt{false}, $\boldsymbol{\pi}_{h,u}^m, \pi_{h,p}^m, I_{\text{iters}}, \|\boldsymbol{b}\|_\infty$ \tcp*{\textsf{Catastrophic OSGS Divergence Abort}}
\end{algorithm}

\section{Structural Orchestration and Convergence Analysis}

The solver pipeline rigorously separates pure Newton-Raphson mathematics from physical boundary analysis through a cascading fallback strategy. By explicitly monitoring algorithmic exit bounds, the system aggressively tests fast evaluations while preserving mathematical robustness across dimensionless parameter extrema (e.g., $Re, Da \sim 10^6$).

\subsection{Algorithmic Fallback Progression}
Robustness is achieved by evaluating nonlinear failures geometrically rather than relying exclusively on continuous float exceptions. The progression orchestrates the solvers as follows:
\begin{enumerate}
    \item \textbf{Aggressive Exact Newton Initialization (\hyperref[alg:A]{Algorithm A}):} The solver unconditionally attempts to root the initial guess directly inside the quadratic convergence basin using pure analytic derivatives. The internal logic delegates the mathematical updates to linear resolution (\textbf{Algorithm A.1}), safe bounded fractional steps (\textbf{Algorithm A.2}), and strict termination guards (\textbf{Algorithm A.3}).
    \item \textbf{Picard Globalization Fallback (\hyperref[alg:B]{Algorithm B}):} If \textbf{Algorithm A} aborts aggressively due to linesearch depletion evaluated via \textbf{Algorithm A.3} (\textsf{Linesearch Failed}) or chaotic divergence (\textsf{Merit Divergence Escaped}), the sequence retreats securely. It temporarily abandons exact Jacobian symmetry and invokes classical Picard iterations. Because the Picard Jacobian actively drops the advective coupling tensor, it stabilizes chaotic boundaries, smoothing the global fields strictly into the local quadratic basin before dynamically re-engaging \textbf{Algorithm A} seamlessly to securely polish the algebraic residual.
    \subitem \textit{Dynamic Adaptivity:} The structural limits mapping the Picard configuration unconditionally adapt structurally evaluating the convection Reynolds bounds ($Re$) and reaction Damköhler extents ($Da$). Highly convective geometry thresholds ($Re \geq 10^4$) explicitly expand iterative smoothing limits dynamically allowing sufficient boundary layer relaxation preventing subsequent Newton convergence breakdown arrays.
\end{enumerate}

\subsection{Adaptive Spatial \(\tau_{\text{ftol}}\) Constraint Constraints}
In continuous parameter extraction runs, maintaining rigidly static algebraic stopping conditions structurally disrupts fine-mesh evaluations. The orchestration pipeline continuously scales $\tau_{\text{ftol}}$ conditionally proportionate directly against the projected analytical error variance: 
\begin{equation}
    \tau_{\text{ftol, dynamic}} = \max\left(\tau_{\text{ftol, static}}, \; \min(C_{\text{ceiling}}, \; C_{\text{sf}} \times h^{k+1})\right)
\end{equation}
Where $h$ represents the linear block resolution scalar and $k$ represents the velocity finite element interpolant order. Natively adapting these limits using the bounding scalar $C_{\text{sf}}$ (spatial safety factor) ensures the non-linear algebraic root is consistently evaluated proportionally beneath topological precision barriers ($C_{\text{sf}} \times h^{k+1}$). The absolute constraint ceiling $C_{\text{ceiling}}$ intrinsically proves algorithmic execution avoids catastrophically locking the non-linear matrices natively seeking functionally impossible precision topologies bounds on ultra-coarse grid spans.

\subsection{Dynamic Stagnation bounds and Mathematical Truthfulness}
As established in \textbf{Algorithm A.3}, the $\tau_{\text{stagnation}}$ check functions explicitly as an absolute terminating safety net rather than an unconditional shortcut. The algebraic noise floor is strictly verified \textit{only} if a catastrophic structural exit (\textsf{Linesearch Failed}, \textsf{Iteration Cap}) terminates the sequence gracefully prior to mathematically striking the $\tau_{\text{ftol}}$ bound identically.

\textbf{False Equivalency Failsafe:} It is mathematically critical to logically decouple \textit{noise floor saturation} from \textit{step exhaustion}. If the inner solver terminates natively via iteration budget depletion (\textsf{Iteration Cap Stagnation}) or extreme coordinate stalling (\textsf{X-Tol Stagnation}) while tracking an algebraic residual strictly above the dynamic noise floor ($\|\boldsymbol{b}\|_\infty \gg \tau_{\text{stagnation}}$), the pipeline classifies the state as an unconditional numerical failure. The orchestration framework explicitly forbids coercing unconditionally exhausted sequences into "pseudo-converged" approvals, natively protecting the staggering fractional structures from bootstrapping completely unresolved mathematical topologies.

Because classical double precision noise floors explicitly skew dynamically as fine mesh matrices scale topologically (condition parameter $K \approx \mathcal{O}(h^{-2})$), the orchestration logic natively maps dynamic extrapolation extracting algebraic condition limits bounding against target physical norms ($\tau_{\text{stagnation}} \propto 10^{-16} N_{\text{dofs}}^2 \max(1, Re)$). Natively ensuring that these bounds remain distinctly superior linearly to $\tau_{\text{ftol}}$ rigorously prevents the fine-mesh algorithms from aborting physically viable solution sequences prematurely.

\subsection{Fractional Verification Requirements}
Instead of implicitly treating fractional evaluation success as mathematically rigorous, \textbf{Algorithm C (OSGS)} binds independent mathematical criteria:
A fractional update sweep is deemed converged exclusively when \textit{either} or \textit{both} the spatial $L^\infty(\Omega)$ coordinate drift ($d_{\text{state}}^m$) \textit{and/or} orthogonal projection vectors ($d_\pi^m$) independently hit continuous equilibrium $\tau_{\text{OSGS}}$. If the inner linear fractional solver mapped via \textbf{Algorithm A} collapses (\textsf{Linesearch Failed}), the staggered loop abruptly halts instantly recording a native fraction abort rather than falsely reporting stable limits conditionally.

\subsection{OSGS Fractional Relaxation (\hyperref[alg:C]{Algorithm C}) Mechanics}
